00049 // This class is designed to be very small (usually 8 butes) and make no memory
00050 // allocation.
00051 // It only works for keywords less than 256 char long and with list of less than
00052 // 256 keywords
00053 //
00054 //
00055 // The keywords* needs to point to a `constexpr` array of alphabetically sorted
00056 // `StringMatcherAction`. The last action should have the `match` field set to
00057 // `nullptr`. Its value is the default value that will be returned when we
00058 // aren't matching anything...
00059 template <typename T, const StringMatcherAction<T> *keywords>
00060 class StringMatcher {
00061     static constexpr size_t totlen_ = GetNumActions(keywords);
00062     static_assert(totlen_ > 0, "Too few kwds...");
00063     static_assert(totlen_ < 256, "Too many kwds...");
00064     static_assert(GetMaxMatch(keywords) < 256,
00065                   "One of the kwds is too long...");
00066 
00067     uint8_t pos_ = 0;  // We could figure out how many bytes we need for that by
00068                        // finding the length of the longest string
00069 
00070     uint8_t min_ = 0;  // If we have more that 255 keywords we should use a
00071                        // larger type. we could figure this out at compile time
00072                        // by looking at totlen_
00073     uint8_t max_ = totlen_ - 1;
00074     enum State : uint8_t { kRunning, kSuccess, kFailed } state_ = kRunning;
00075 
00076 public:
00077     // Advance the matcher by one character
00078     void AddChar(const char c_in) {
00079         switch (state_) {
00080             case kFailed:
00081                 return;
00082             case kSuccess:
00083                 assert(false);
00084                 return;
00085             case kRunning:
00086                 break;
00087         }
00088 
00089         while (min_ < max_ && keywords[min_].match[pos_] != c_in) {
00090             ++min_;
00091         }
00092 
00093         while (min_ < max_ && keywords[max_].match[pos_] != c_in) {
00094             --max_;
00095         }
00096 
00097         if (min_ == max_) {
00098             if (keywords[max_].match[pos_] != c_in) {
00099                 state_ = kFailed;
00100                 return;
00101             } else if (c_in == '\000') {
00102                 state_ = kSuccess;
00103                 return;
00104             }
00105         }
00106         ++pos_;
00107     }
00108 
00109     T GetResult() const {
00110         return keywords[(state_ == kSuccess) ? min_ : totlen_].val;
00111     }
00112 
00113     void Reset() {
00114         state_ = kRunning;
00115         pos_ = 0;
00116         min_ = 0;
00117         max_ = totlen_ - 1;
00118     }
00119 };
